    data.forEach(function(p){
        who.forEach(function(d){
            var out = {};
            out.candidate = d;
            out.number = p[d] === '--' ? 0 : parseFloat(p[d]);
            out.poll = p.Poll;
            out.size = Math.random();
            out.when = p.Date;
            polls.push(out);
