        chart_type: 'histogram',
        width: 600,
        height: 300,

        bar_margin: 0,
        bins: 150,
        target: '#ufos',
        y_extended_ticks: true,
        mouseover: function(d, i) {
            var string;
            if (d.x < 12) {
                string = d3.round(d.x,2) + ' Months';
